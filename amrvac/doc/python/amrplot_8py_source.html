438  self.n=np.empty(2)
439  self.n[0] = (self.bob[0]-self.alice[0])/np.sqrt((self.bob[0]-self.alice[0])**2+(self.bob[1]-self.alice[1])**2)
440  self.n[1] = (self.bob[1]-self.alice[1])/np.sqrt((self.bob[0]-self.alice[0])**2+(self.bob[1]-self.alice[1])**2)
441  self.epsilon = 2.e-1

444  def run(self):
445 
446  self.x=copy.deepcopy(self.alice)
447  i=0
448  while (self.n[0]*(self.bob[0]-self.x[0])+self.n[1]*(self.bob[1]-self.x[1]))>0 and i<self.stepmax:
449  myIcell = self.data.getIcellByPoint(self.x[0],self.x[1])
450  self.icells.append(myIcell)
451  self.step(myIcell)
452  i=i+1

455  centerpoints=self.data.getCenterPoints()
456  vert=self.data.getVert(icell)
457  delta = np.min([np.abs(vert[0][0]-vert[1][0]),np.abs(vert[0][1]-vert[3][1])])/2.
458  delta = delta * (1.-self.epsilon)
459  myIcell=icell

462  myIcell=self.data.getIcellByPoint(self.x[0],self.x[1])
